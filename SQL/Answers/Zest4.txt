SELECT o.id "Nr", 
c.name "Klient", 
p.name "Produkt", 
o.payment_type "Platnoœæ", 
o.date_ordered "Data zamówienia", 
i.price "Cena", 
i.quantity "Liczba"
FROM
ord o, product p, customer c, item i
WHERE 
o.sales_rep_id = c.sales_rep_id AND
o.id=i.ord_id AND
i.product_id=p.id AND 
o.payment_type like 'CASH' AND
o.date_ordered LIKE '92/09/%' AND
o.customer_id=c.id
ORDER BY o.id , p.name;

c.name "Klient", 
o.payment_type "Platnoœæ", 
o.date_ordered "Data zamówienia", 
i.price*i.quantity "Wysokoœæ zamówienia"
FROM
ord o, product p, customer c, item i
WHERE 
o.sales_rep_id = c.sales_rep_id AND
o.id=i.ord_id AND
i.product_id=p.id AND 
o.payment_type LIKE 'CASH' AND
o.date_ordered LIKE '92/09/%' AND
o.customer_id=c.id
ORDER BY o.id;
